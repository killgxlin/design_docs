struct client_t {
    tcp::socket socket;
    bool destroying;
    bool valid;

    // worker
    // socket
    start_recv();

    // worker
    // start_recv
    handle_recv();

    // sender
    // socket
    start_send();

    // worker
    handle_send();
};

struct network_t {
    queue destroy_queue;
    stack free_stack;
    vector all_vector;

    thread_group worker; // destroy_queue, free_stack
    thread destroyer; // destroy_queue, free_stack
    thread sender; // all_vector, destroy_queue
    thread_group others; // destroy_queue

    // worker, destroyer, others
    // free_stack
    start_accept(); 

    // worker
    // 
    handle_accept(); 

    // worker, others
    // destroy_queue
    disconnect();
};
